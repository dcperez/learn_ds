116    /**
117       A visitor whose visit method is called for each visited node
118       during a tree traversal.
119    */
120    public interface Visitor
121    {
122       /**
123          This method is called for each visited node.
124          @param data the data of the node
125       */
126       void visit(Object data);
127    }
128 
129    /**
130       Traverses this tree in preorder.
131       @param v the visitor to be invoked at each node
132    */
133    public void preorder(Visitor v) { preorder(root, v); }
134    
135    /**
136       Traverses the tree with a given root in preorder.
137       @param n the root of the tree
138       @param v the visitor to be invoked at each node
139    */
140    private static void preorder(Node n, Visitor v)
141    {
142       if (n == null) { return; }
143       v.visit(n.data);
144       for (Node c : n.children) { preorder(c, v); }
145    }
146 
147    /**
148       Traverses this tree in postorder.
149       @param v the visitor to be invoked at each node
150    */
151    public void postorder(Visitor v) { postorder(root, v); }
152    
153    /**
154       Traverses the tree with a given root in postorder.
155       @param n the root of the tree
156       @param v the visitor to be invoked at each node
157    */
158    private static void postorder(Node n, Visitor v)
159    {
160       if (n == null) { return; }
161       for (Node c : n.children) { postorder(c, v); }
162       v.visit(n.data);
163    }

165    /**
166       This iterator visits the nodes of a tree in 
167       breadth-first order.
168    */
169    class BreadthFirstIterator implements Iterator
170    {
171       private Queue<Node> q;
172 
173       /**
174          Constructs an iterator for a given tree.
175          @param root the root of the tree
176       */
177       public BreadthFirstIterator(Node root)
178       {
179          q = new LinkedList<>();
180          if (root != null) { q.add(root); }
181       }
182 
183       public boolean hasNext() { return q.size() > 0; }
184 
185       public Object next() 
186       {
187          Node n = q.remove();
188          for (Node c : n.children) { q.add(c); }
189          return n.data;
190       }
191 
192       public void remove() { throw new UnsupportedOperationException(); }
193    }
194 
195    public Iterator iterator() 
196    {
197       return new BreadthFirstIterator(root);
198    }
